import shader from "./inlined/shader";
import THREE from "three";

/**
 * The Water class handles reflection and refraction 
 * processing and provides a material to use on 
 * arbitrary meshes.
 *
 * @class Water
 * @constructor
 * @param {Object} [options] - The options.
 * @param {Texture} [options.normalMap] - The normalmap to use for the waves.
 * @param {Vector3} [options.lightPosition] - The position of the main light source.
 * @param {Boolean} [options.lowQuality=false] - The quality of the shader.
 */

export function Water(options) {

	if(options === undefined) { options = {}; }

	/**
	 * The water material.
	 *
	 * @property material
	 * @type {ShaderMaterial}
	 */

	this.material = new THREE.ShaderMaterial({

		uniforms: {

			logDepthBufFC: {type: "f", value: 0.0},

			fogDensity: {type: "f", value: 0.0},
			fogNear: {type: "f", value: 1.0},
			fogFar: {type: "f", value: 2000.0},
			fogColor: {type: "c", value: new THREE.Color(0xffffff)},

			time: {type: "f", value: 0.0},
			visibility: {type: "f", value: 28.0},

			reflectionMap: {type: "t", value: null},
			refractionMap: {type: "t", value: null},
			normalMap: {type: "t", value: options.normalMap},
			deathMap: {type: "t", value: options.depthMap},
			offsetRepeat: {type: "v4", value: new THREE.Vector4(0.0, 0.0, 1.0, 1.0)},

			waveScale: {type: "f", value: 350.0},
			waveChoppiness: {type: "f", value: 0.15},
			bigWaves: {type: "v2", value: new THREE.Vector2(0.3, 0.3)},
			midWaves: {type: "v2", value: new THREE.Vector2(0.3, 0.15)},
			smallWaves: {type: "v2", value: new THREE.Vector2(0.15, 0.1)},

			windSpeed: {type: "f", value: 0.2},
			windDirection: {type: "v2", value: new THREE.Vector2(0.5, -0.8)},
			lightPosition: {type: "v3", value: options.lightPosition},

			waterDensity: {type: "f", value: 1.0},
			chromaticAberration: {type: "f", value: 0.002},
			waterBump: {type: "f", value: 1.0},
			reflectionBump: {type: "f", value: 0.04},
			refractionBump: {type: "f", value: 0.03},
			eta: {type: "f", value: 1.33},

			waterColor: {type: "v4", value: new THREE.Vector4(0.2, 0.4, 0.5, 1.0)},
			sunSpecular: {type: "f", value: 1000.0},
			scatterAmount: {type: "f", value: 3.5},
			scatterColor: {type: "v4", value: new THREE.Vector4(0.0, 1.0, 0.95, 1.0)},

			luminosity: {type: "v3", value: new THREE.Vector3(0.3, 0.59, 0.11)},
			mudExtinction: {type: "v3", value: new THREE.Vector3(1.0, 0.7, 0.5)},
			sunExtinction: {type: "v3", value: new THREE.Vector3(0.45, 0.55, 0.68)},
			waterExtinction: {type: "v3", value: new THREE.Vector3(0.6, 0.8, 1.0)}

		},

		fragmentShader: options.lowQuality ? shader.fragment.low : shader.fragment.high,
		vertexShader: shader.vertex,

		derivatives: true,
		fog: true

	});

}
